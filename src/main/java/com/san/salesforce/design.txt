public Response<Json> getConfig(String configId) {
        SafeCheck and sanitize the payload String configId
        RedisResponse rs = RedisDao.get(configId);
        if(rs==null) {
            throw new InvalidRequestParameters("Invalid configId given");
        }
}

path("/config/")
public Response<Json> saveConfigKey(String key, Strign Object, int ttl) {
        SafeCheck and sanitize the payload String configId
        //auto expire time. TTL
        RedisResponse rs = RedisDao.get(configId);
        if(rs==null) {
            //It is not expired
            rs.save(key, value, ttl);
        }
}
